List<Person> persons = Arrays.asList(
   new  Person("Max", 18");
   new Person("tony",22);

List<Person> filtered = 
   persons
      .stream()
      .filter(p -> p.name.startsWith("M"))
      .collect(Collectors.toList());
System.out.println(filtered);
